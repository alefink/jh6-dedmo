23      * REST controller for managing {@link org.jhipster.blog.domain.Category}.

25     @RestController
26     @RequestMapping("/api")
27     public class CategoryResource {
28     
29         private final Logger log = LoggerFactory.getLogger(CategoryResource.class);
30     
31         private static final String ENTITY_NAME = "category";
32     
33         @Value("${jhipster.clientApp.name}")
34         private String applicationName;
35     
36         private final CategoryService categoryService;
37     
38         public CategoryResource(CategoryService categoryService) {
39             this.categoryService = categoryService;
40         }

42         /**
43          * {@code POST  /categories} : Create a new category.
44          *
45          * @param category the category to create.
46          * @return the {@link ResponseEntity} with status {@code 201 (Created)} and with body the new category, or with status {@code 400 (Bad Request)} if the category has already an ID.
47          * @throws URISyntaxException if the Location URI syntax is incorrect.
48          */
49         @PostMapping("/categories")
50         public ResponseEntity<Category> createCategory(@Valid @RequestBody Category category) throws URISyntaxException {
51             log.debug("REST request to save Category : {}", category);
52             if (category.getId() != null) {
53                 throw new BadRequestAlertException("A new category cannot already have an ID", ENTITY_NAME, "idexists");
54             }
55             Category result = categoryService.save(category);
56             return ResponseEntity.created(new URI("/api/categories/" + result.getId()))
57                 .headers(HeaderUtil.createEntityCreationAlert(applicationName, true, ENTITY_NAME, result.getId().toString()))
58                 .body(result);
59         }
60     
61         /**
62          * {@code PUT  /categories} : Updates an existing category.
63          *
64          * @param category the category to update.
65          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the updated category,
66          * or with status {@code 400 (Bad Request)} if the category is not valid,
67          * or with status {@code 500 (Internal Server Error)} if the category couldn't be updated.
68          * @throws URISyntaxException if the Location URI syntax is incorrect.
69          */
70         @PutMapping("/categories")
71         public ResponseEntity<Category> updateCategory(@Valid @RequestBody Category category) throws URISyntaxException {
72             log.debug("REST request to update Category : {}", category);
73             if (category.getId() == null) {
74                 throw new BadRequestAlertException("Invalid id", ENTITY_NAME, "idnull");
75             }
76             Category result = categoryService.save(category);
77             return ResponseEntity.ok()
78                 .headers(HeaderUtil.createEntityUpdateAlert(applicationName, true, ENTITY_NAME, category.getId().toString()))
79                 .body(result);
80         }
81     
82         /**
83          * {@code GET  /categories} : get all the categories.
84          *
85          * @param eagerload flag to eager load entities from relationships (This is applicable for many-to-many).
86          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and the list of categories in body.
87          */
88         @GetMapping("/categories")
89         public List<Category> getAllCategories(@RequestParam(required = false, defaultValue = "false") boolean eagerload) {
90             log.debug("REST request to get all Categories");
91             return categoryService.findAll();
92         }
93     
94         /**
95          * {@code GET  /categories/:id} : get the "id" category.
96          *
97          * @param id the id of the category to retrieve.
98          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the category, or with status {@code 404 (Not Found)}.
99          */
100         @GetMapping("/categories/{id}")
101         public ResponseEntity<Category> getCategory(@PathVariable Long id) {
102             log.debug("REST request to get Category : {}", id);
103             Optional<Category> category = categoryService.findOne(id);
104             return ResponseUtil.wrapOrNotFound(category);
105         }
106     
107         /**
108          * {@code DELETE  /categories/:id} : delete the "id" category.
109          *
110          * @param id the id of the category to delete.
111          * @return the {@link ResponseEntity} with status {@code 204 (NO_CONTENT)}.
112          */
113         @DeleteMapping("/categories/{id}")
114         public ResponseEntity<Void> deleteCategory(@PathVariable Long id) {
115             log.debug("REST request to delete Category : {}", id);
116             categoryService.delete(id);
117             return ResponseEntity.noContent().headers(HeaderUtil.createEntityDeletionAlert(applicationName, true, ENTITY_NAME, id.toString())).build();
118         }
