90         public User registerUser(UserDTO userDTO, String password) {
91             userRepository.findOneByLogin(userDTO.getLogin().toLowerCase()).ifPresent(existingUser -> {
92                 boolean removed = removeNonActivatedUser(existingUser);
93                 if (!removed) {
94                     throw new LoginAlreadyUsedException();
95                 }
96             });
97             userRepository.findOneByEmailIgnoreCase(userDTO.getEmail()).ifPresent(existingUser -> {
98                 boolean removed = removeNonActivatedUser(existingUser);
99                 if (!removed) {
100                     throw new EmailAlreadyUsedException();
101                 }
102             });
103             User newUser = new User();
104             String encryptedPassword = passwordEncoder.encode(password);
105             newUser.setLogin(userDTO.getLogin().toLowerCase());
106             // new user gets initially a generated password
107             newUser.setPassword(encryptedPassword);
108             newUser.setFirstName(userDTO.getFirstName());
109             newUser.setLastName(userDTO.getLastName());
110             newUser.setEmail(userDTO.getEmail().toLowerCase());
111             newUser.setImageUrl(userDTO.getImageUrl());
112             newUser.setLangKey(userDTO.getLangKey());
113             // new user is not active
114             newUser.setActivated(false);
115             // new user gets registration key
116             newUser.setActivationKey(RandomUtil.generateActivationKey());
117             Set<Authority> authorities = new HashSet<>();
118             authorityRepository.findById(AuthoritiesConstants.USER).ifPresent(authorities::add);
119             newUser.setAuthorities(authorities);
120             userRepository.save(newUser);
121             this.clearUserCaches(newUser);
122             log.debug("Created Information for User: {}", newUser);
123             return newUser;
124         }

126         private boolean removeNonActivatedUser(User existingUser){
127             if (existingUser.getActivated()) {
128                  return false;
129             }
130             userRepository.delete(existingUser);
131             userRepository.flush();
132             this.clearUserCaches(existingUser);
133             return true;
134         }

136         public User createUser(UserDTO userDTO) {
137             User user = new User();
138             user.setLogin(userDTO.getLogin().toLowerCase());
139             user.setFirstName(userDTO.getFirstName());
140             user.setLastName(userDTO.getLastName());
141             user.setEmail(userDTO.getEmail().toLowerCase());
142             user.setImageUrl(userDTO.getImageUrl());
143             if (userDTO.getLangKey() == null) {
144                 user.setLangKey(Constants.DEFAULT_LANGUAGE); // default language
145             } else {
146                 user.setLangKey(userDTO.getLangKey());
147             }
148             String encryptedPassword = passwordEncoder.encode(RandomUtil.generatePassword());
149             user.setPassword(encryptedPassword);
150             user.setResetKey(RandomUtil.generateResetKey());
151             user.setResetDate(Instant.now());
152             user.setActivated(true);
153             if (userDTO.getAuthorities() != null) {
154                 Set<Authority> authorities = userDTO.getAuthorities().stream()
155                     .map(authorityRepository::findById)
156                     .filter(Optional::isPresent)
157                     .map(Optional::get)
158                     .collect(Collectors.toSet());
159                 user.setAuthorities(authorities);
160             }
161             userRepository.save(user);
162             this.clearUserCaches(user);
163             log.debug("Created Information for User: {}", user);
164             return user;
165         }

190         /**
191          * Update all information for a specific user, and return the modified user.
192          *
193          * @param userDTO user to update.
194          * @return updated user.
195          */
196         public Optional<UserDTO> updateUser(UserDTO userDTO) {
197             return Optional.of(userRepository
198                 .findById(userDTO.getId()))
199                 .filter(Optional::isPresent)
200                 .map(Optional::get)
201                 .map(user -> {
202                     this.clearUserCaches(user);
203                     user.setLogin(userDTO.getLogin().toLowerCase());
204                     user.setFirstName(userDTO.getFirstName());
205                     user.setLastName(userDTO.getLastName());
206                     user.setEmail(userDTO.getEmail().toLowerCase());
207                     user.setImageUrl(userDTO.getImageUrl());
208                     user.setActivated(userDTO.isActivated());
209                     user.setLangKey(userDTO.getLangKey());
210                     Set<Authority> managedAuthorities = user.getAuthorities();
211                     managedAuthorities.clear();
212                     userDTO.getAuthorities().stream()
213                         .map(authorityRepository::findById)
214                         .filter(Optional::isPresent)
215                         .map(Optional::get)
216                         .forEach(managedAuthorities::add);
217                     this.clearUserCaches(user);
218                     log.debug("Changed Information for User: {}", user);
219                     return user;
220                 })
221                 .map(UserDTO::new);
222         }

232         public void changePassword(String currentClearTextPassword, String newPassword) {
233             SecurityUtils.getCurrentUserLogin()
234                 .flatMap(userRepository::findOneByLogin)
235                 .ifPresent(user -> {
236                     String currentEncryptedPassword = user.getPassword();
237                     if (!passwordEncoder.matches(currentClearTextPassword, currentEncryptedPassword)) {
238                         throw new InvalidPasswordException();
239                     }
240                     String encryptedPassword = passwordEncoder.encode(newPassword);
241                     user.setPassword(encryptedPassword);
242                     this.clearUserCaches(user);
243                     log.debug("Changed password for User: {}", user);
244                 });
245         }

293             Objects.requireNonNull(cacheManager.getCache(UserRepository.USERS_BY_LOGIN_CACHE)).evict(user.getLogin());
