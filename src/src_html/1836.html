78             UndertowServletWebServerFactory container = new UndertowServletWebServerFactory();
79             webConfigurer.customize(container);
80             assertThat(container.getMimeMappings().get("abs")).isEqualTo("audio/x-mpeg");
81             assertThat(container.getMimeMappings().get("html")).isEqualTo("text/html;charset=utf-8");
82             assertThat(container.getMimeMappings().get("json")).isEqualTo("text/html;charset=utf-8");
83             if (container.getDocumentRoot() != null) {
