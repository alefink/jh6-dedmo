36     /**
37      * REST controller for managing users.
38      * <p>
39      * This class accesses the {@link User} entity, and needs to fetch its collection of authorities.
40      * <p>
41      * For a normal use-case, it would be better to have an eager relationship between User and Authority,
42      * and send everything to the client side: there would be no View Model and DTO, a lot less code, and an outer-join
43      * which would be good for performance.
44      * <p>
45      * We use a View Model and a DTO for 3 reasons:
46      * <ul>
47      * <li>We want to keep a lazy association between the user and the authorities, because people will
48      * quite often do relationships with the user, and we don't want them to get the authorities all
49      * the time for nothing (for performance reasons). This is the #1 goal: we should not impact our users'
50      * application because of this use-case.</li>
51      * <li> Not having an outer join causes n+1 requests to the database. This is not a real issue as
52      * we have by default a second-level cache. This means on the first HTTP call we do the n+1 requests,
53      * but then all authorities come from the cache, so in fact it's much better than doing an outer join
54      * (which will get lots of data from the database, for each HTTP call).</li>
55      * <li> As this manages users, for security reasons, we'd rather have a DTO layer.</li>
56      * </ul>
57      * <p>
58      * Another option would be to have a specific JPA entity graph to handle this case.
59      */

82         /**
83          * {@code POST  /users}  : Creates a new user.
84          * <p>
85          * Creates a new user if the login and email are not already used, and sends an
86          * mail with an activation link.
87          * The user needs to be activated on creation.
88          *
89          * @param userDTO the user to create.
90          * @return the {@link ResponseEntity} with status {@code 201 (Created)} and with body the new user, or with status {@code 400 (Bad Request)} if the login or email is already in use.
91          * @throws URISyntaxException if the Location URI syntax is incorrect.
92          * @throws BadRequestAlertException {@code 400 (Bad Request)} if the login or email is already in use.
93          */
94         @PostMapping("/users")
95         @PreAuthorize("hasRole(\"" + AuthoritiesConstants.ADMIN + "\")")
96         public ResponseEntity<User> createUser(@Valid @RequestBody UserDTO userDTO) throws URISyntaxException {
97             log.debug("REST request to save User : {}", userDTO);
98     
99             if (userDTO.getId() != null) {
100                 throw new BadRequestAlertException("A new user cannot already have an ID", "userManagement", "idexists");
101                 // Lowercase the user login before comparing with database
102             } else if (userRepository.findOneByLogin(userDTO.getLogin().toLowerCase()).isPresent()) {
103                 throw new LoginAlreadyUsedException();
104             } else if (userRepository.findOneByEmailIgnoreCase(userDTO.getEmail()).isPresent()) {
105                 throw new EmailAlreadyUsedException();
106             } else {
107                 User newUser = userService.createUser(userDTO);
108                 mailService.sendCreationEmail(newUser);
109                 return ResponseEntity.created(new URI("/api/users/" + newUser.getLogin()))
110                     .headers(HeaderUtil.createAlert(applicationName,  "userManagement.created", newUser.getLogin()))
111                     .body(newUser);
112             }
113         }

115         /**
116          * {@code PUT /users} : Updates an existing User.
117          *
118          * @param userDTO the user to update.
119          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the updated user.
120          * @throws EmailAlreadyUsedException {@code 400 (Bad Request)} if the email is already in use.
121          * @throws LoginAlreadyUsedException {@code 400 (Bad Request)} if the login is already in use.
122          */
123         @PutMapping("/users")
124         @PreAuthorize("hasRole(\"" + AuthoritiesConstants.ADMIN + "\")")
125         public ResponseEntity<UserDTO> updateUser(@Valid @RequestBody UserDTO userDTO) {
126             log.debug("REST request to update User : {}", userDTO);
127             Optional<User> existingUser = userRepository.findOneByEmailIgnoreCase(userDTO.getEmail());
128             if (existingUser.isPresent() && (!existingUser.get().getId().equals(userDTO.getId()))) {
129                 throw new EmailAlreadyUsedException();
130             }
131             existingUser = userRepository.findOneByLogin(userDTO.getLogin().toLowerCase());
132             if (existingUser.isPresent() && (!existingUser.get().getId().equals(userDTO.getId()))) {
133                 throw new LoginAlreadyUsedException();
134             }
135             Optional<UserDTO> updatedUser = userService.updateUser(userDTO);
136     
137             return ResponseUtil.wrapOrNotFound(updatedUser,
138                 HeaderUtil.createAlert(applicationName, "userManagement.updated", userDTO.getLogin()));
139         }

141         /**
142          * {@code GET /users} : get all users.
143          *
144          * @param queryParams a {@link MultiValueMap} query parameters.
145          * @param uriBuilder a {@link UriComponentsBuilder} URI builder.
146          * @param pageable the pagination information.
147          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body all users.
148          */
149         @GetMapping("/users")
150         public ResponseEntity<List<UserDTO>> getAllUsers(@RequestParam MultiValueMap<String, String> queryParams, UriComponentsBuilder uriBuilder, Pageable pageable) {
151             final Page<UserDTO> page = userService.getAllManagedUsers(pageable);
152             HttpHeaders headers = PaginationUtil.generatePaginationHttpHeaders(uriBuilder.queryParams(queryParams), page);
153             return new ResponseEntity<>(page.getContent(), headers, HttpStatus.OK);
154         }
