92             Map<String, Object> data = new HashMap<>();
93             StringBuilder largeData = new StringBuilder();
94             for (int i = 0; i < EVENT_DATA_COLUMN_MAX_LENGTH + 10; i++) {
95                 largeData.append("a");
96             }
97             data.put("test-key", largeData);
98             AuditEvent event = new AuditEvent("test-user", "test-type", data);
99             customAuditEventRepository.add(event);
100             List<PersistentAuditEvent> persistentAuditEvents = persistenceAuditEventRepository.findAll();
101             assertThat(persistentAuditEvents).hasSize(1);
102             PersistentAuditEvent persistentAuditEvent = persistentAuditEvents.get(0);
103             assertThat(persistentAuditEvent.getPrincipal()).isEqualTo(event.getPrincipal());
104             assertThat(persistentAuditEvent.getAuditEventType()).isEqualTo(event.getType());
105             assertThat(persistentAuditEvent.getData()).containsKey("test-key");
106             String actualData = persistentAuditEvent.getData().get("test-key");
107             assertThat(actualData.length()).isEqualTo(EVENT_DATA_COLUMN_MAX_LENGTH);
108             assertThat(actualData).isSubstringOf(largeData);
109             assertThat(persistentAuditEvent.getAuditEventDate().truncatedTo(ChronoUnit.MILLIS))
110                 .isEqualTo(event.getTimestamp().truncatedTo(ChronoUnit.MILLIS));
