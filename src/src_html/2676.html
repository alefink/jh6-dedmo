29      * Controller advice to translate the server side exceptions to client-friendly json structures.
30      * The error response follows RFC7807 - Problem Details for HTTP APIs (https://tools.ietf.org/html/rfc7807).
31      */
32     @ControllerAdvice
33     public class ExceptionTranslator implements ProblemHandling, SecurityAdviceTrait {
34     
35         private static final String FIELD_ERRORS_KEY = "fieldErrors";
36         private static final String MESSAGE_KEY = "message";
37         private static final String PATH_KEY = "path";
38         private static final String VIOLATIONS_KEY = "violations";
39     
40         @Value("${jhipster.clientApp.name}")
41         private String applicationName;
42     
43         /**
44          * Post-process the Problem payload to add the message key for the front-end if needed.
45          */
46         @Override
47         public ResponseEntity<Problem> process(@Nullable ResponseEntity<Problem> entity, NativeWebRequest request) {
48             if (entity == null) {
49                 return entity;
50             }
51             Problem problem = entity.getBody();
52             if (!(problem instanceof ConstraintViolationProblem || problem instanceof DefaultProblem)) {
53                 return entity;
54             }
55             ProblemBuilder builder = Problem.builder()
56                 .withType(Problem.DEFAULT_TYPE.equals(problem.getType()) ? ErrorConstants.DEFAULT_TYPE : problem.getType())
57                 .withStatus(problem.getStatus())
58                 .withTitle(problem.getTitle())
59                 .with(PATH_KEY, request.getNativeRequest(HttpServletRequest.class).getRequestURI());
60     
61             if (problem instanceof ConstraintViolationProblem) {
62                 builder
63                     .with(VIOLATIONS_KEY, ((ConstraintViolationProblem) problem).getViolations())
64                     .with(MESSAGE_KEY, ErrorConstants.ERR_VALIDATION);
65             } else {
66                 builder
67                     .withCause(((DefaultProblem) problem).getCause())
68                     .withDetail(problem.getDetail())
69                     .withInstance(problem.getInstance());
70                 problem.getParameters().forEach(builder::with);
71                 if (!problem.getParameters().containsKey(MESSAGE_KEY) && problem.getStatus() != null) {
72                     builder.with(MESSAGE_KEY, "error.http." + problem.getStatus().getStatusCode());
73                 }
74             }
75             return new ResponseEntity<>(builder.build(), entity.getHeaders(), entity.getStatusCode());
76         }

78         @Override
79         public ResponseEntity<Problem> handleMethodArgumentNotValid(MethodArgumentNotValidException ex, @Nonnull NativeWebRequest request) {
80             BindingResult result = ex.getBindingResult();
81             List<FieldErrorVM> fieldErrors = result.getFieldErrors().stream()
82                 .map(f -> new FieldErrorVM(f.getObjectName(), f.getField(), f.getCode()))
83                 .collect(Collectors.toList());
84     
85             Problem problem = Problem.builder()
86                 .withType(ErrorConstants.CONSTRAINT_VIOLATION_TYPE)
87                 .withTitle("Method argument not valid")
88                 .withStatus(defaultConstraintViolationStatus())
89                 .with(MESSAGE_KEY, ErrorConstants.ERR_VALIDATION)
90                 .with(FIELD_ERRORS_KEY, fieldErrors)
91                 .build();
92             return create(ex, problem, request);
93         }
94     
95         @ExceptionHandler
96         public ResponseEntity<Problem> handleNoSuchElementException(NoSuchElementException ex, NativeWebRequest request) {
97             Problem problem = Problem.builder()
98                 .withStatus(Status.NOT_FOUND)
99                 .with(MESSAGE_KEY, ErrorConstants.ENTITY_NOT_FOUND_TYPE)
100                 .build();
101             return create(ex, problem, request);
102         }
103     
104         @ExceptionHandler
105         public ResponseEntity<Problem> handleBadRequestAlertException(BadRequestAlertException ex, NativeWebRequest request) {
106             return create(ex, request, HeaderUtil.createFailureAlert(applicationName, true, ex.getEntityName(), ex.getErrorKey(), ex.getMessage()));
107         }
108     
109         @ExceptionHandler
110         public ResponseEntity<Problem> handleConcurrencyFailure(ConcurrencyFailureException ex, NativeWebRequest request) {
111             Problem problem = Problem.builder()
112                 .withStatus(Status.CONFLICT)
113                 .with(MESSAGE_KEY, ErrorConstants.ERR_CONCURRENCY_FAILURE)
114                 .build();
115             return create(ex, problem, request);
116         }
