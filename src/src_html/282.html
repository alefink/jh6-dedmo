207         @Test
208         public void testSendLocalizedEmailForAllSupportedLanguages() throws Exception {
209             User user = new User();
210             user.setLogin("john");
211             user.setEmail("john.doe@example.com");
212             for (String langKey : languages) {
213                 user.setLangKey(langKey);
214                 mailService.sendEmailFromTemplate(user, "mail/testEmail", "email.test.title");
215                 verify(javaMailSender, atLeastOnce()).send(messageCaptor.capture());
216                 MimeMessage message = messageCaptor.getValue();
217     
218                 String propertyFilePath = "i18n/messages_" + getJavaLocale(langKey) + ".properties";
219                 URL resource = this.getClass().getClassLoader().getResource(propertyFilePath);
220                 File file = new File(new URI(resource.getFile()).getPath());
221                 Properties properties = new Properties();
222                 properties.load(new InputStreamReader(new FileInputStream(file), Charset.forName("UTF-8")));
223     
224                 String emailTitle = (String) properties.get("email.test.title");
225                 assertThat(message.getSubject()).isEqualTo(emailTitle);
226                 assertThat(message.getContent().toString()).isEqualToNormalizingNewlines("<html>" + emailTitle + ", http://127.0.0.1:8080, john</html>\n");
227             }
228         }

230         /**
231          * Convert a lang key to the Java locale.
232          */
233         private String getJavaLocale(String langKey) {
234             String javaLangKey = langKey;
235             Matcher matcher2 = PATTERN_LOCALE_2.matcher(langKey);
236             if (matcher2.matches()) {
237                 javaLangKey = matcher2.group(1) + "_"+ matcher2.group(2).toUpperCase();
238             }
239             Matcher matcher3 = PATTERN_LOCALE_3.matcher(langKey);
240             if (matcher3.matches()) {
241                 javaLangKey = matcher3.group(1) + "_" + matcher3.group(2) + "_" + matcher3.group(3).toUpperCase();
242             }
243             return javaLangKey;
244         }
