25      * REST controller for managing {@link org.jhipster.blog.domain.Blog}.

27     @RestController

33         private static final String ENTITY_NAME = "blog";

35         @Value("${jhipster.clientApp.name}")
36         private String applicationName;

38         private final BlogRepository blogRepository;
39     
40         public BlogResource(BlogRepository blogRepository) {

44         /**
45          * {@code POST  /blogs} : Create a new blog.
46          *
47          * @param blog the blog to create.
48          * @return the {@link ResponseEntity} with status {@code 201 (Created)} and with body the new blog, or with status {@code 400 (Bad Request)} if the blog has already an ID.
49          * @throws URISyntaxException if the Location URI syntax is incorrect.
50          */
51         @PostMapping("/blogs")
52         public ResponseEntity<?> createBlog(@Valid @RequestBody Blog blog) throws URISyntaxException {
53             log.debug("REST request to save Blog : {}", blog);
54             if (blog.getId() != null) {
55                 throw new BadRequestAlertException("A new blog cannot already have an ID", ENTITY_NAME, "idexists");
56             }
57             if (!blog.getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
58                 return new ResponseEntity<>("error.http.403", HttpStatus.FORBIDDEN);
59             }
60             Blog result = blogRepository.save(blog);
61             return ResponseEntity.created(new URI("/api/blogs/" + result.getId()))
62                 .headers(HeaderUtil.createEntityCreationAlert(applicationName, true, ENTITY_NAME, result.getId().toString()))
63                 .body(result);
64         }
65     
66         /**
67          * {@code PUT  /blogs} : Updates an existing blog.
68          *
69          * @param blog the blog to update.
70          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the updated blog,
71          * or with status {@code 400 (Bad Request)} if the blog is not valid,
72          * or with status {@code 500 (Internal Server Error)} if the blog couldn't be updated.
73          * @throws URISyntaxException if the Location URI syntax is incorrect.
74          */
75         @PutMapping("/blogs")
76         public ResponseEntity<?> updateBlog(@Valid @RequestBody Blog blog) throws URISyntaxException {
77             log.debug("REST request to update Blog : {}", blog);
78             if (blog.getId() == null) {
79                 throw new BadRequestAlertException("Invalid id", ENTITY_NAME, "idnull");
80             }
81             if (blog.getUser() != null &&
82                 !blog.getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
83                 return new ResponseEntity<>("error.http.403", HttpStatus.FORBIDDEN);
84             }
85             Blog result = blogRepository.save(blog);
86             return ResponseEntity.ok()
87                 .headers(HeaderUtil.createEntityUpdateAlert(applicationName, true, ENTITY_NAME, blog.getId().toString()))
88                 .body(result);
89         }

91         /**
92          * {@code GET  /blogs} : get all the blogs.
93          *
94          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and the list of blogs in body.
95          */
96         @GetMapping("/blogs")
97         public List<Blog> getAllBlogs() {
98             log.debug("REST request to get all Blogs");
99             return blogRepository.findByUserIsCurrentUser();
100         }
101     
102         /**
103          * {@code GET  /blogs/:id} : get the "id" blog.
104          *
105          * @param id the id of the blog to retrieve.
106          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the blog, or with status {@code 404 (Not Found)}.
107          */
108         @GetMapping("/blogs/{id}")
109         public ResponseEntity<?> getBlog(@PathVariable Long id) {
110             log.debug("REST request to get Blog : {}", id);
111             Optional<Blog> blog = blogRepository.findById(id);
112             if (blog.isPresent() && blog.get().getUser() != null &&
113                 !blog.get().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
114                 return new ResponseEntity<>("error.http.403", HttpStatus.FORBIDDEN);
115             }
116             return ResponseUtil.wrapOrNotFound(blog);
117         }
118     
119         /**
120          * {@code DELETE  /blogs/:id} : delete the "id" blog.
121          *
122          * @param id the id of the blog to delete.
123          * @return the {@link ResponseEntity} with status {@code 204 (NO_CONTENT)}.
124          */
125         @DeleteMapping("/blogs/{id}")
126         public ResponseEntity<?> deleteBlog(@PathVariable Long id) {
127             log.debug("REST request to delete Blog : {}", id);
128             Optional<Blog> blog = blogRepository.findById(id);
129             if (blog.isPresent() && blog.get().getUser() != null &&
130                 !blog.get().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
131                 return new ResponseEntity<>("error.http.403", HttpStatus.FORBIDDEN);
132             }
133             blogRepository.deleteById(id);
134             return ResponseEntity.noContent().headers(HeaderUtil.createEntityDeletionAlert(applicationName, true, ENTITY_NAME, id.toString())).build();
135         }
