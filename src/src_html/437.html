50         /**
51          * {@code POST  /entries} : Create a new entry.
52          *
53          * @param entry the entry to create.
54          * @return the {@link ResponseEntity} with status {@code 201 (Created)} and with body the new entry, or with status {@code 400 (Bad Request)} if the entry has already an ID.
55          * @throws URISyntaxException if the Location URI syntax is incorrect.
56          */
57         @PostMapping("/entries")
58         public ResponseEntity<?> createEntry(@Valid @RequestBody Entry entry) throws URISyntaxException {
59             log.debug("REST request to save Entry : {}", entry);
60             if (entry.getId() != null) {
61                 throw new BadRequestAlertException("A new entry cannot already have an ID", ENTITY_NAME, "idexists");
62             }
63             Entry result = entryRepository.save(entry);
64             if (entry.getBlog() != null &&
65                 !entry.getBlog().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
66                 return new ResponseEntity<>("Unauthorized", HttpStatus.UNAUTHORIZED);
67             }
68             return ResponseEntity.created(new URI("/api/entries/" + result.getId()))
69                 .headers(HeaderUtil.createEntityCreationAlert(applicationName, true, ENTITY_NAME, result.getId().toString()))
70                 .body(result);
71         }

73         /**
74          * {@code PUT  /entries} : Updates an existing entry.
75          *
76          * @param entry the entry to update.
77          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the updated entry,
78          * or with status {@code 400 (Bad Request)} if the entry is not valid,
79          * or with status {@code 500 (Internal Server Error)} if the entry couldn't be updated.
80          * @throws URISyntaxException if the Location URI syntax is incorrect.
81          */
82         @PutMapping("/entries")
83         public ResponseEntity<?> updateEntry(@Valid @RequestBody Entry entry) throws URISyntaxException {
84             log.debug("REST request to update Entry : {}", entry);
85             if (entry.getId() == null) {
86                 throw new BadRequestAlertException("Invalid id", ENTITY_NAME, "idnull");
87             }
88             if (entry.getBlog() != null &&
89                 !entry.getBlog().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
90                 return new ResponseEntity<>("Unauthorized", HttpStatus.UNAUTHORIZED);
91             }
92             Entry result = entryRepository.save(entry);
93             return ResponseEntity.ok()
94                 .headers(HeaderUtil.createEntityUpdateAlert(applicationName, true, ENTITY_NAME, entry.getId().toString()))
95                 .body(result);
96         }

98         /**
99          * {@code GET  /entries} : get all the entries.
100          *
101          * @param pageable the pagination information.
102          * @param queryParams a {@link MultiValueMap} query parameters.
103          * @param uriBuilder a {@link UriComponentsBuilder} URI builder.
104          * @param eagerload flag to eager load entities from relationships (This is applicable for many-to-many).
105          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and the list of entries in body.
106          */
107         @GetMapping("/entries")
108         public ResponseEntity<List<Entry>> getAllEntries(Pageable pageable, @RequestParam MultiValueMap<String, String> queryParams, UriComponentsBuilder uriBuilder, @RequestParam(required = false, defaultValue = "false") boolean eagerload) {
109             log.debug("REST request to get a page of Entries");
110             Page<Entry> page = entryRepository.findByBlogUserLoginOrderByDateDesc(SecurityUtils.getCurrentUserLogin().orElse(null), pageable);
111             HttpHeaders headers = PaginationUtil.generatePaginationHttpHeaders(uriBuilder.queryParams(queryParams), page);
112             return ResponseEntity.ok().headers(headers).body(page.getContent());
113         }

115         /**
116          * {@code GET  /entries/:id} : get the "id" entry.
117          *
118          * @param id the id of the entry to retrieve.
119          * @return the {@link ResponseEntity} with status {@code 200 (OK)} and with body the entry, or with status {@code 404 (Not Found)}.
120          */
121         @GetMapping("/entries/{id}")
122         public ResponseEntity<?> getEntry(@PathVariable Long id) {
123             log.debug("REST request to get Entry : {}", id);
124             Optional<Entry> entry = entryRepository.findOneWithEagerRelationships(id);
125             if (entry.isPresent() && entry.get().getBlog() != null &&
126                 !entry.get().getBlog().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
127                 return new ResponseEntity<>("Unauthorized", HttpStatus.UNAUTHORIZED);
128             }
129             return ResponseUtil.wrapOrNotFound(entry);
130         }

133          * {@code DELETE  /entries/:id} : delete the "id" entry.
134          *
135          * @param id the id of the entry to delete.
136          * @return the {@link ResponseEntity} with status {@code 204 (NO_CONTENT)}.
137          */
138         @DeleteMapping("/entries/{id}")
139         public ResponseEntity<?> deleteEntry(@PathVariable Long id) {
140             log.debug("REST request to delete Entry : {}", id);
141             Optional<Entry> entry = entryRepository.findOneWithEagerRelationships(id);
142             if (entry.isPresent() && entry.get().getBlog() != null &&
143                 !entry.get().getBlog().getUser().getLogin().equals(SecurityUtils.getCurrentUserLogin().orElse(""))) {
144                 return new ResponseEntity<>("Unauthorized", HttpStatus.UNAUTHORIZED);
145             }
146             entryRepository.deleteById(id);
147             return ResponseEntity.noContent().headers(HeaderUtil.createEntityDeletionAlert(applicationName, true, ENTITY_NAME, id.toString())).build();
148         }
