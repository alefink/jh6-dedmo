52       protected convertDateFromClient(category: ICategory): ICategory {
53         const copy: ICategory = Object.assign({}, category, {
54           dateAdded: category.dateAdded != null && category.dateAdded.isValid() ? category.dateAdded.format(DATE_FORMAT) : null,
55           dateModified: category.dateModified != null && category.dateModified.isValid() ? category.dateModified.format(DATE_FORMAT) : null
56         });
57         return copy;
58       }

60       protected convertDateFromServer(res: EntityResponseType): EntityResponseType {
61         if (res.body) {
62           res.body.dateAdded = res.body.dateAdded != null ? moment(res.body.dateAdded) : null;
63           res.body.dateModified = res.body.dateModified != null ? moment(res.body.dateModified) : null;
64         }
65         return res;
66       }

68       protected convertDateArrayFromServer(res: EntityArrayResponseType): EntityArrayResponseType {

72             category.dateModified = category.dateModified != null ? moment(category.dateModified) : null;
