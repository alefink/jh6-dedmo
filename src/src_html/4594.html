87       setFileData(event, field: string, isImage) {
88         return new Promise((resolve, reject) => {
89           if (event && event.target && event.target.files && event.target.files[0]) {
90             const file = event.target.files[0];
91             if (isImage && !/^image\//.test(file.type)) {
92               reject(`File was expected to be an image but was found to be ${file.type}`);
93             } else {
94               const filedContentType: string = field + 'ContentType';
95               this.dataUtils.toBase64(file, base64Data => {
96                 this.editForm.patchValue({
97                   [field]: base64Data,
98                   [filedContentType]: file.type
99                 });
100               });
101             }
102           } else {
103             reject(`Base64 data was not set as file could not be extracted from passed parameter: ${event}`);
104           }
105         }).then(
