15     @Injectable({ providedIn: 'root' })
16     export class ProductResolve implements Resolve<IProduct> {
17       constructor(private service: ProductService) {}
18     
19       resolve(route: ActivatedRouteSnapshot, state: RouterStateSnapshot): Observable<IProduct> {
20         const id = route.params['id'] ? route.params['id'] : null;
21         if (id) {
22           return this.service.find(id).pipe(
23             filter((response: HttpResponse<Product>) => response.ok),
24             map((product: HttpResponse<Product>) => product.body)
25           );
26         }
27         return of(new Product());
28       }
29     }
