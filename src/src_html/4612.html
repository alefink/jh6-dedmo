36         return this.http

39       }

44           .get<IProduct[]>(this.resourceUrl, { params: options, observe: 'response' })

52       protected convertDateFromClient(product: IProduct): IProduct {
53         const copy: IProduct = Object.assign({}, product, {
54           dateAdded: product.dateAdded != null && product.dateAdded.isValid() ? product.dateAdded.format(DATE_FORMAT) : null,
55           dateModified: product.dateModified != null && product.dateModified.isValid() ? product.dateModified.format(DATE_FORMAT) : null
56         });
57         return copy;
58       }
59     
60       protected convertDateFromServer(res: EntityResponseType): EntityResponseType {
61         if (res.body) {
62           res.body.dateAdded = res.body.dateAdded != null ? moment(res.body.dateAdded) : null;
63           res.body.dateModified = res.body.dateModified != null ? moment(res.body.dateModified) : null;
64         }
65         return res;
66       }
67     
68       protected convertDateArrayFromServer(res: EntityArrayResponseType): EntityArrayResponseType {
69         if (res.body) {
70           res.body.forEach((product: IProduct) => {
71             product.dateAdded = product.dateAdded != null ? moment(product.dateAdded) : null;
72             product.dateModified = product.dateModified != null ? moment(product.dateModified) : null;
73           });
74         }
